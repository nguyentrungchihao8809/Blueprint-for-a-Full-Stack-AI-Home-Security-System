#include <LiquidCrystal.h>
#include <DHT.h>

// Cấu hình chân kết nối LCD 16x2
const uint8_t LCD_RS = 12;
const uint8_t LCD_EN = 11;
const uint8_t LCD_D4 = 7;
const uint8_t LCD_D5 = 6;
const uint8_t LCD_D6 = 5;
const uint8_t LCD_D7 = 4;
LiquidCrystal lcd(LCD_RS, LCD_EN, LCD_D4, LCD_D5, LCD_D6, LCD_D7);

// Cấu hình chân cảm biến và cơ cấu chấp hành
#define DHT_PIN 2
#define DHT_TYPE DHT11
DHT dht(DHT_PIN, DHT_TYPE);

#define GAS_PIN A0
#define PIR_PIN 3        
#define DOOR_PIN 8      
#define LED_PIN 9
#define BUZZER_PIN 10

// Ngưỡng cảnh báo an toàn
const float CRITICAL_TEMP = 75.0;
const int CRITICAL_GAS = 300;

// Biến điều khiển thủ công từ Gateway
bool manualLED = false;
bool manualBuzzer = false;

// Biến quản lý thời gian quét định kỳ
unsigned long previousMillis = 0;   
const long INTERVAL = 2500;  

// Biến lưu trạng thái trước đó để chống trùng lặp sự kiện
int lastPirState = LOW;
int lastDoorState = HIGH;

// Biến lưu trữ giá trị cảm biến toàn cục
float currentTemp = 0.0;
float currentHumid = 0.0;

void setup() {
  Serial.begin(9600);
  while (!Serial) { ; }
  
  lcd.begin(16, 2);
  lcd.print("Smart Home AI");
  lcd.setCursor(0, 1);
  lcd.print("System Ready...");
  
  pinMode(PIR_PIN, INPUT);
  pinMode(DOOR_PIN, INPUT_PULLUP); 
  pinMode(LED_PIN, OUTPUT);
  pinMode(BUZZER_PIN, OUTPUT);
  
  dht.begin();
  delay(2000); 
  lcd.clear();
}

void loop() {
  // 1. XỬ LÝ LỆNH ĐIỀU KHIỂN TỪ GATEWAY (PYTHON)
  if (Serial.available() > 0) {
    char receivedChar = Serial.read(); 
    
    switch (receivedChar) {
      case '1':
        manualLED = true;
        Serial.println("EVENT:LED_MANUAL_ON");
        break;
      case '2':
        manualLED = false;
        Serial.println("EVENT:LED_MANUAL_OFF");
        break;
      case '3':
        manualBuzzer = true;  
        Serial.println("EVENT:BUZZER_MANUAL_ON");
        break;
      case '4':
        manualBuzzer = false; 
        Serial.println("EVENT:BUZZER_MANUAL_OFF");
        break;
      default:
        break;
    }
  }

  // 2. ĐỌC DỮ LIỆU CẢM BIẾN THỜI GIAN THỰC
  int pirState = digitalRead(PIR_PIN);
  int doorState = digitalRead(DOOR_PIN);
  int gasAnalog = analogRead(GAS_PIN);
  int gasPPM = map(gasAnalog, 0, 1023, 0, 1000);
  
  // Kiểm tra điều kiện tự động kích hoạt báo động (Hỏa hoạn/Rò rỉ khí)
  bool isFireDanger = (currentTemp > CRITICAL_TEMP || gasPPM > CRITICAL_GAS);

  // Điều khiển thiết bị đầu ra (Ưu tiên chế độ tự động an toàn trước điều khiển thủ công)
  digitalWrite(LED_PIN, (isFireDanger || manualLED) ? HIGH : LOW);
  digitalWrite(BUZZER_PIN, (isFireDanger || manualBuzzer) ? HIGH : LOW);

  // 3. GIÁM SÁT VÀ PHÁT HIỆN SỰ KIỆN KHẨN CẤP (NGẮT CHỐNG SPAM)
  if (pirState == HIGH && lastPirState == LOW) {
    Serial.println("EVENT:MOTION_DETECTED");
  }
  lastPirState = pirState;

  if (doorState == HIGH && lastDoorState == LOW) {
    Serial.println("EVENT:DOOR_OPENED");
  }
  lastDoorState = doorState;

  // 4. CẬP NHẬT ĐỊNH KỲ LCD VÀ TRUYỀN DỮ LIỆU LÊN GATEWAY
  unsigned long currentMillis = millis(); 
  if (currentMillis - previousMillis >= INTERVAL) {
    previousMillis = currentMillis; 

    // Đọc và lọc nhiễu dữ liệu từ cảm biến DHT
    float rawHumid = dht.readHumidity();
    float rawTemp = dht.readTemperature();
    
    if (!isnan(rawTemp) && rawTemp > -40.0 && rawTemp < 125.0) {
      currentTemp = rawTemp;
    }
    if (!isnan(rawHumid)) {
      currentHumid = rawHumid;
    }

    // Hiển thị cục bộ trên LCD 16x2
    lcd.setCursor(0, 0);
    lcd.print("T:"); lcd.print((int)currentTemp); lcd.print("C ");
    lcd.print("H:"); lcd.print((int)currentHumid); lcd.print("%   ");
    
    lcd.setCursor(0, 1);
    lcd.print("G:"); lcd.print(gasPPM); lcd.print("PPM ");
    lcd.print(pirState == HIGH ? "M:!" : "M:-");
    lcd.print(doorState == HIGH ? " D:O" : " D:C");

    // Đóng gói và gửi chuỗi dữ liệu chuẩn lên Python Gateway
    Serial.print("TEMP:");   Serial.print(currentTemp);
    Serial.print("|HUMID:"); Serial.print(currentHumid);
    Serial.print("|GAS:");   Serial.print(gasPPM);
    Serial.print("|PIR:");   Serial.print(pirState);
    Serial.print("|DOOR:");  Serial.println(doorState); 
  }
}
